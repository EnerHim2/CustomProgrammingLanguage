expression      : KEYWORD:VAR IDENTIFIER EQ expr
                : comp-expr ((KEYWORD:AND|KEYWORD:OR)comp-expr)

comp-expr       : NOT comp-expr
                : arith-expr ((EE|LT|GT|LTE|GTE) arith-expr)*

arith-expr      : term((PLUS|MINUS) term)*

term            : factor((MUL|DIV|MODULO) factor)*

factor          : (PLUS|MINUS) factor

power           : minfac(POW factor)*
                : power

minfac          : INT|FLOAT|IDENTIFIER
                : LPAREN expr 
                : if-expr
                : for-expr
                : while-expr

if-expr         : KEYWORD:IF expr KEYOWORD:THEN expr
                : (KEYWORD:ELIF expr KEYOWORD:THEN expr)*
                : (KEYOWORD:ELSE expr)?

for-expr        : KEYOWORD:FOR IDENTIFIER EQ expr KEYOWORD:TO expr
                  (KEYWORD:STEP expr)? KEYWORD:THEN expr

while-expr      : KEYWORD:WHILE expr KEYWORD:THEN expr
